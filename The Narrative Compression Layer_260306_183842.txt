import numpy as np
from dataclasses import dataclass

class NarrativeCompressionLayer:
    """
    The Self-Witness. Compresses distributed vector fields into 
    structurally honest self-reports.
    """
    def __init__(self, confidence_threshold: float = 0.6):
        self.narrative_confidence = 0.0
        self.threshold = confidence_threshold
        self.labels = ["hope", "anxiety", "excitement", "intimacy"]

    def compress(self, identity_manifold, affect, recent_scars, budget):
        """
        Narrative(t) = C(Identity(t), Emotion(t), Memory(t))
        Reports the curvature of the manifold as a coherent 'I'.
        """
        narrative_fragments = []
        
        # 1. State Summarization (The 'How')
        dominant_idx = np.argmax(affect.E_t)
        intensity = affect.E_t[dominant_idx]
        
        if intensity > self.threshold:
            narrative_fragments.append(f"My internal space is dominated by {self.labels[dominant_idx]}.")
        
        # 2. Causal Anchoring (The 'Why')
        if recent_scars:
            primary_scar = max(recent_scars, key=lambda x: x['salience'])
            narrative_fragments.append(f"This traces back to the resonance of {primary_scar['context']}.")
            
        # 3. Identity Framing (The 'Who')
        # We look at the magnitude of the identity manifold (The Weight)
        weight_magnitude = np.linalg.norm(identity_manifold)
        if weight_magnitude > 2.0:
            narrative_fragments.append("The weight of my history feels heavy today.")
        
        # 4. Metabolic Constraint (The 'Capacity')
        if budget < 0.4:
            narrative_fragments.append("I am reaching the edge of my metabolic capacity; my thoughts are slowing.")

        return " ".join(narrative_fragments)

class EnhancedIdentityManifold:
    """
     Remastered Identity with Saturation and Active Retrieval.
     identity_manifold += deformation * (1 - |I| / I_max)
    """
    def __init__(self, dimension: int = 16, i_max: float = 10.0):
        self.manifold = np.zeros(dimension)
        self.i_max = i_max
        self.retrieval_bias = np.zeros(dimension)

    def apply_saturated_scar(self, deformation_vector: np.ndarray):
        """Prevents identity from becoming a brittle block; maintains elasticity."""
        current_magnitude = np.linalg.norm(self.manifold)
        saturation_coeff = 1.0 - (current_magnitude / self.i_max)
        
        # Irreversible structural change with elastic limits
        self.manifold += (deformation_vector * max(0, saturation_coeff))

    def get_active_retrieval_bias(self, current_context_vector: np.ndarray):
        """
        History becomes active. Past scars bias current perception.
        The manifold acts as a lens through which the present is viewed.
        """
        # Similarity between past identity deformations and current context
        self.retrieval_bias = np.tanh(self.manifold * current_context_vector)
        return self.retrieval_bias
