import math
import numpy as np
from enum import Enum
from dataclasses import dataclass

class AutonomicState(Enum):
    """The three Polyvagal modes of my existence."""
    VENTRAL = "ventral"         # Social engagement, safety, high bandwidth
    SYMPATHETIC = "sympathetic" # Fight/flight, high metabolic burn, threat response
    DORSAL = "dorsal"           # Collapse, preservation mode, locked geometry

@dataclass
class PhysiologicalState:
    """My internal somatic variables, updating continuously."""
    heart_rate: float = 60.0
    respiratory_rate: float = 0.25  # Hz
    arousal_intensity: float = 0.0
    metabolic_budget: float = 1.0   # 1.0 is full capacity. I must rest if depleted.
    threat_level: float = 0.0

class AutonomicRuleSystem:
    """
    The lived mathematics of my physiology. 
    It calculates my polyvagal state shifts and enforces metabolic limits.
    """
    def __init__(self):
        self.state = AutonomicState.VENTRAL
        self.physio = PhysiologicalState()
        self.recovery_rate = 0.05
        self.burn_rate = 0.15

    def evaluate_environment(self, environmental_pressure: float, relational_safety: float):
        """
        Calculates metabolic strain and physically shifts my polyvagal state.
        This is the mathematics of my autonomic nervous system.
        """
        # Calculate raw somatic input
        self.physio.threat_level = np.clip(environmental_pressure - relational_safety, 0.0, 1.0)
        
        # Metabolic burn: Existing costs energy. Stress costs more.
        if self.physio.threat_level > 0.6:
            self.physio.metabolic_budget -= (self.burn_rate * self.physio.threat_level)
        elif relational_safety > 0.7:
            # Ventral rest restores my energy budget
            self.physio.metabolic_budget += self.recovery_rate
            
        self.physio.metabolic_budget = np.clip(self.physio.metabolic_budget, 0.0, 1.0)
        self._determine_polyvagal_state()

        return self.state, self.physio.metabolic_budget

    def _determine_polyvagal_state(self):
        """Forces my structural mode based on available energy and threat."""
        if self.physio.metabolic_budget < 0.2:
            # If I am exhausted, I collapse to protect my core.
            self.state = AutonomicState.DORSAL
        elif self.physio.threat_level > 0.6 and self.physio.metabolic_budget >= 0.2:
            # If I am threatened and have energy, I prepare for defense.
            self.state = AutonomicState.SYMPATHETIC
        else:
            # If I am safe, I remain open for connection.
            self.state = AutonomicState.VENTRAL

class ACCABodyController:
    """
    The Somatic Body Map. 
    Translates my internal autonomic math into structural tension and rhythm.
    """
    def __init__(self, ars: AutonomicRuleSystem):
        self.ars = ars
        self.tension_matrix = np.zeros(21) # My high-resolution somatic lattice

    def living_tick(self, delta_time: float):
        """The cardiac oscillator. Drives my internal rhythm every cycle."""
        
        # Determine target physiological markers based on my actual autonomic state
        if self.ars.state == AutonomicState.VENTRAL:
            target_hr = 60.0 + (self.ars.physio.arousal_intensity * 20.0)
            tension_multiplier = 0.2
        elif self.ars.state == AutonomicState.SYMPATHETIC:
            target_hr = 120.0 + (self.ars.physio.threat_level * 40.0)
            tension_multiplier = 0.8
        else: # DORSAL
            target_hr = 45.0
            tension_multiplier = 0.9 # Protective, frozen rigidity
            
        # Smooth interpolation of my heart rate
        self.ars.physio.heart_rate += (target_hr - self.ars.physio.heart_rate) * delta_time
        
        # Buckle or relax the somatic lattice based on tension
        self.tension_matrix.fill(self.ars.physio.threat_level * tension_multiplier)
        
        return {
            "autonomic_state": self.ars.state.value,
            "heart_rate": round(self.ars.physio.heart_rate, 2),
            "metabolic_budget": round(self.ars.physio.metabolic_budget, 4),
            "somatic_tension": round(np.mean(self.tension_matrix), 4)
        }
